  <script>
    // Исходные данные (пусто при загрузке)
    let logData = [];

    // Элементы DOM
    const fileInput = document.getElementById('file-input');
    const actionSelect = document.getElementById('action-select');
    const timeSelect = document.getElementById('time-select');
    const outputText = document.getElementById('output-text');
    const copyToast = document.getElementById('copy-toast');

    // --- Toast уведомление ---
    let toastTimeout = null;
    function showToast(message, bgColor = '#4caf50') {
      copyToast.textContent = message;
      copyToast.style.backgroundColor = bgColor;
      copyToast.classList.add('show');
      if (toastTimeout) clearTimeout(toastTimeout);
      toastTimeout = setTimeout(() => {
        copyToast.classList.remove('show');
      }, 2500);
    }

    // --- Обработка файла ---
    fileInput.addEventListener('change', (e) => {
      const file = e.target.files[0];
      if (!file) return;
      const reader = new FileReader();
      reader.onload = () => {
        const text = reader.result;
        parseFileContent(text);
        showToast('Файл успешно загружен', '#4caf50');
      };
      reader.onerror = () => {
        showToast('Ошибка при чтении файла', '#e53935');
      };
      reader.readAsText(file);
    });

    // --- Парсинг содержимого файла ---
    function parseFileContent(text) {
      logData = text.split('\n').map(line => {
        const match = line.match(/^(\d+)\. \| (\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) (.*)$/);
        if (!match) return null;
        return { entry_number: match[1], timestamp: match[2], action: match[3].trim() };
      }).filter(Boolean);
      applyFilters();
    }

    // --- Фильтрация и вывод ---
    function applyFilters() {
      const actionFilter = actionSelect.value;
      const timeFilter = timeSelect.value;

      // Вычисляем дату отсева по времени
      let cutoffDate = null;
      const now = new Date();
      if (timeFilter === 'week') {
        cutoffDate = new Date(now.getTime() - 7*24*60*60*1000);
      } else if (timeFilter === '2weeks') {
        cutoffDate = new Date(now.getTime() - 14*24*60*60*1000);
      } else if (timeFilter === 'month') {
        cutoffDate = new Date(now.getTime() - 30*24*60*60*1000);
      }

      // Фильтрация
      const filtered = logData.filter(entry => {
        if (cutoffDate && new Date(entry.timestamp) < cutoffDate) return false;
        if (actionFilter === 'Все действия') return true;
        return entry.action === actionFilter;
      });

      // Вывод
      outputText.innerHTML = '';
      filtered.forEach(entry => {
        const span = document.createElement('span');
        span.textContent = `${entry.entry_number}. | ${entry.timestamp} ${entry.action}`;
        span.title = 'Нажмите, чтобы скопировать';
        span.addEventListener('click', () => {
          navigator.clipboard.writeText(span.textContent)
            .then(() => showToast('Текст скопирован', '#2196f3'))
            .catch(() => showToast('Ошибка копирования', '#e53935'));
        });
        outputText.appendChild(span);
      });
    }

    actionSelect.addEventListener('change', applyFilters);
    timeSelect.addEventListener('change', applyFilters);

    // --- Автоматическое обновление с GitHub ---
    const GITHUB_RAW_URL = 'https://raw.githubusercontent.com/filedelete/Arzabcheck/refs/heads/main/abcheck.txt';

    let githubFileContent = null;

    async function fetchGitHubFile() {
      try {
        const response = await fetch(GITHUB_RAW_URL + '?t=' + Date.now());
        if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
        const text = await response.text();

        if (githubFileContent === null) {
          githubFileContent = text;
          if (logData.length === 0) {
            parseFileContent(text);
            showToast('Данные с GitHub успешно загружены', '#4caf50');
          }
        } else if (githubFileContent !== text) {
          githubFileContent = text;
          parseFileContent(text);
          showToast('Обновления с GitHub успешно применены', '#4caf50');
        }
      } catch (error) {
        console.error('Ошибка при загрузке файла с GitHub:', error);
        showToast('Ошибка загрузки обновлений с GitHub', '#e53935');
      }
    }

    // Загрузка при старте
    fetchGitHubFile();

    // Проверка обновлений каждые 5 минут
    setInterval(fetchGitHubFile, 300000);
  </script>
