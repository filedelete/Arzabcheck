  <script>
    const fileInput = document.getElementById('file-input');
    const actionSelect = document.getElementById('action-select');
    const timeSelect = document.getElementById('time-select');
    const outputText = document.getElementById('output-text');
    const toggleThemeBtn = document.getElementById('toggle-theme');
    const copyToast = document.getElementById('copy-toast');

    const updateNotification = document.getElementById('update-notification');
    const btnUpdateYes = document.getElementById('btn-update-yes');
    const btnUpdateNo = document.getElementById('btn-update-no');

    let logData = [];
    let isDarkTheme = true;

    const actionMap = [
      { keyword: "уволил игрока", label: "Уволил игрока" },
      { keyword: "подтверждает участие", label: "Подтверждает участие на мероприятие фракции" },
      { keyword: "изменил ранг игрока", label: "Изменил ранг игрока" },
      { keyword: "установил игроку", label: "Установил игроку тег" },
      { keyword: "принял игрока", label: "Принял игрока в организацию" },
      { keyword: "открыл общак", label: "Открыл склад организации" },
      { keyword: "закрыл общак", label: "Закрыл склад организации" },
      { keyword: "дал выговор", label: "Дал выговор игроку" },
      { keyword: "снял выговор", label: "Снял выговор с игрока" },
      { keyword: "пополнил счет", label: "Пополнил счет организации" },
      { keyword: "назначил собеседование", label: "Назначил собеседование" }
    ];

    toggleThemeBtn.addEventListener('click', () => {
      isDarkTheme = !isDarkTheme;
      if (isDarkTheme) {
        document.body.classList.remove('light-theme');
      } else {
        document.body.classList.add('light-theme');
      }
    });

    fileInput.addEventListener('change', (e) => {
      const file = e.target.files[0];
      if (!file) return;
      const reader = new FileReader();
      reader.onload = (e) => {
        const text = e.target.result;
        logData = text.split('\n').filter(line => line.trim() !== '');
        renderOutput();
      };
      reader.readAsText(file);
    });

    actionSelect.addEventListener('change', renderOutput);
    timeSelect.addEventListener('change', renderOutput);

    function filterLogs() {
      const actionFilter = actionSelect.value;
      const timeFilter = timeSelect.value;
      const now = new Date();
      let cutoffDate;

      switch (timeFilter) {
        case 'week':
          cutoffDate = new Date(now.getTime() - 7*24*60*60*1000);
          break;
        case '2weeks':
          cutoffDate = new Date(now.getTime() - 14*24*60*60*1000);
          break;
        case '3weeks':
          cutoffDate = new Date(now.getTime() - 21*24*60*60*1000);
          break;
        case 'month':
          cutoffDate = new Date(now.getTime() - 30*24*60*60*1000);
          break;
        default:
          cutoffDate = new Date(0);
      }

      return logData.filter(line => {
        // Пример формата лога: "[2025-05-15 12:34:56] ... действие ..."
        // Попытаемся получить дату из строки:
        const dateMatch = line.match(/\[(\d{4}-\d{2}-\d{2})\s[\d:]+\]/);
        if (!dateMatch) return false;
        const date = new Date(dateMatch[1]);
        if (date < cutoffDate) return false;

        if (actionFilter === 'Все действия') return true;

        // Для фильтра по действию - ищем ключевые слова:
        const actionObj = actionMap.find(a => a.label === actionFilter);
        if (!actionObj) return true;

        return line.toLowerCase().includes(actionObj.keyword);
      });
    }

    function renderOutput() {
      const filtered = filterLogs();
      if (filtered.length === 0) {
        outputText.textContent = 'Нет записей для выбранных фильтров.';
        return;
      }

      // Добавим анимацию появления по строкам
      outputText.innerHTML = '';
      filtered.forEach((line, i) => {
        const span = document.createElement('span');
        span.textContent = line;
        span.style.animationDelay = (i * 0.05) + 's';
        span.addEventListener('click', () => {
          copyToClipboard(line);
          highlightLine(span);
        });
        outputText.appendChild(span);
      });
    }

    function copyToClipboard(text) {
      navigator.clipboard.writeText(text).then(() => {
        showCopyToast();
      });
    }

    function showCopyToast() {
      copyToast.classList.add('show');
      setTimeout(() => copyToast.classList.remove('show'), 1500);
    }

    let lastActiveSpan = null;
    function highlightLine(span) {
      if (lastActiveSpan) lastActiveSpan.classList.remove('active');
      span.classList.add('active');
      lastActiveSpan = span;
    }

    // Проверка обновлений с GitHub (пример)
    // Поменяйте URL на ваш raw файл из репозитория GitHub
    const GITHUB_RAW_URL = 'https://raw.githubusercontent.com/username/repo/main/logfile.txt';

    async function checkForUpdates() {
      try {
        const response = await fetch(GITHUB_RAW_URL, {cache: "no-cache"});
        if (!response.ok) throw new Error('Ошибка загрузки файла');

        const newData = await response.text();

        if (!logData.length) return; // Если у нас нет данных, не показываем уведомление

        // Простейшее сравнение (можно усложнить по хэшам или версиям)
        if (newData !== logData.join('\n')) {
          showUpdateNotification();
        }
      } catch(e) {
        // Игнорируем ошибки, например, оффлайн
        console.log('Update check failed:', e.message);
      }
    }

    function showUpdateNotification() {
      updateNotification.classList.add('show');
    }

    btnUpdateYes.addEventListener('click', async () => {
      // Перезагрузить файл с GitHub и обновить вывод
      try {
        const response = await fetch(GITHUB_RAW_URL, {cache: "no-cache"});
        const newData = await response.text();
        logData = newData.split('\n').filter(line => line.trim() !== '');
        renderOutput();
        updateNotification.classList.remove('show');
      } catch(e) {
        alert('Не удалось загрузить обновление.');
        updateNotification.classList.remove('show');
      }
    });

    btnUpdateNo.addEventListener('click', () => {
      updateNotification.classList.remove('show');
    });

    // Проверяем обновления каждую минуту
    setInterval(checkForUpdates, 60000);

    // Проверяем обновления при загрузке страницы (если есть данные)
    window.addEventListener('load', () => {
      if (logData.length) checkForUpdates();
    });
  </script>
